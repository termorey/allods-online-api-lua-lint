На главную

avatar.GetVariableInfo( variableId )

Метод возвращает информацию о геймдизайнерской переменной и её значение у главного игрока.

-- библиотека:
avatar

-- объявление:
function GetVariableInfo( variableId )

-- параметры:
variableId: VariableId or string - или идентификатор ресурса переменной или алиас

-- возвращаемые значения:
table or nil - если нет ошибки, то таблица с полями:
  id: VariableId - Id валюты
  sysName: string - внутриигровой псевдоним переменной
  name: WString - название переменной
  sysDebugName: WString - дебажный путь к ресурсу переменной
  value: number - значение переменной
  minValue: number - минимальное значение переменной
  maxValue: number - максимальное значение переменной
  hasRelatedSpell: boolean - переменная влияет на какие-то заклинания игрока (требуется для них)

-- пример:
local ids = avatar.GetVariables()
if ids and ids[0] then
  local info = avatar.GetVariableInfo( ids[0] )
  if info then
    local currentValue = info.value
  end
end
