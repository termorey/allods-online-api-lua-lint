Список рангов достижения.

-- библиотека
medalsLib

-- объявление:
function GetMedalRanks( medalId )

-- параметры
medalId: ObjectId(Number) - идентификатор достижения

-- возвращаемые значения
ranks: table or nil - список рангов, индексация с 1, каждый элемент таблица с полями
  completeProgress: Number(int) - количество очков прогресса необходимых для завершения ранга
  name: WString - название ранга
  description: WString - описание ранга
  image: TextureId - картинка ранга
  score: Number(int) - количество очков за выполнение ранга
  reward: table - награда, таблица с полями
    description: WString - описание награды

-- пример использования
local сategories = medalsLib.GetCategories()
local medalRanks = medalsLib.GetMedalRanks(сategories[0].subCategories[0].medals[0])
