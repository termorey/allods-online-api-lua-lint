На главную

enchant.GetEnchantSlot( itemId )

Возвращает слот энчанта.

Идентификатор энчанта - это идентификатор предмета.

-- библиотека
enchant

-- объявление:
function enchant.GetEnchantSlot( itemId )

-- параметры
itemId: ObjectId - идентификатор энчанта

-- возвращаемые значения
number (enum ENUM_EnchantSlot_...) - слот, в который вставляется энчант

-- пример использования
if enchant.IsEnchant( itemId ) then
  local slot = enchant.GetEnchantSlot( itemId )
end
