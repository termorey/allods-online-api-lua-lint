На главную

wheelOfLoot.GetInfo( id )

Возвращает данные запрашиваемого колеса лута

-- библиотека
wheelOfLoot

-- объявление:
function GetInfo( id )

-- параметры
id: number (int64) - идентификатор колеса лута

-- возвращаемые значения
table of - данные колёса лута
  sysName: string - служебное название
  lifeTime: number - локальное время в ms, когда колесо автоматически разролится
  price: number (integer) - текущая цена разрола
  rollsCount: number (integer) - текущее количество разролов
  maxRollsCount: number (integer) - максимальное количество разролов
  itemPriceId: ObjectId - id предмет, который тратится на разрол
  wheelId: number (int64) - идентификатор колеса лута
  isLoginEvent: boolean - true если это колеса лута логин-ивента
  isManual: boolean - true если награду можно выбирать

-- пример использования
local wheelIds = wheelOfLoot.GetList()
local wheelInfo = wheelOfLoot.GetInfo( wheelIds[0] )
