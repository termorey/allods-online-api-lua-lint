remort.GetAbilitiesForSharing()

Возвращает список умений, которые может расшарить главный игрок для использования своим альтом (реморт).

-- библиотека:
remort

-- объявление:
function GetAbilitiesForSharing()

-- параметры:
нет

-- возвращаемое значение
table: -- список индексированный с 1
  abilityId: AbilityId or nil -- идентификатор ресурса умения
  spellId: SpellId or nil -- идентификатор ресурса заклинания
  isSelected: Boolean -- выбрано(расшарено) в данный момент

-- пример:
local abilities = remort.GetAbilitiesForSharing()
for i, entry in ipairs( abilities ) do
  local abilityId = entry.abilityId
  local abilityInfo = abilityId and avatar.GetAbilityinfo( abilityId ) or {}
  local spellId = entry.spellId
  local spellInfo = spellId and avatar.GetSpellInfo( spellId ) or {}
end
