На главную

remort.GetRemortsList()

Возвращает список альтернативных персонажей, на которых может переключиться главный игрок.

Можно вызывать только после прихода EVENT_REMORT_LIST_GAINED.

-- библиотека:
remort

-- объявление:
function GetRemortsList()

-- параметры:
нет

-- возвращаемое значение
table of table - индексированный с 0 список информации об альтернативных персонажах. Каждая таблица содержит поля:
  name: WString - имя персонажа
  avatarServerId: number (integer) - серверный идентификатор персонажа
  isActive: boolean - это активный в данный момент персонаж
  isPrimary: boolean - это главный персонаж
  level: number (integer) - уровень персонажа
  sysClassName: string or nil - внутрисистемное имя класса персонажа, нелокализуемая строка
  sysRaceName: string or nil - внутрисистемное имя расы персонажа, нелокализуемая строка

-- пример:
if remort.CanGetRemortsList() then
  local remorts = remort.GetRemortsList()
  for i = 0, GetTableSize( remorts - 1 ) do
    local remort = remorts[ i ]
    LogInfo( remort.name )
  end
end
