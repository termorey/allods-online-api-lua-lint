Пользовательские дополнения

На главную

binding.GetSection( sysSectionName )

Возвращает информацию о секции с пользовательскими биндами.

-- библиотека:
binding

-- объявление:
function GetSection( sysSectionName )

-- параметры:
sysSectionName: string - идентификатор секции

-- возвращаемые значения:
таблица с информацией о секции или nil в случае ошибки (отсутствия секции), поля:
 sysName: string - служебное название (идентификатор) секции (то же самое, что и sysSectionName)
 name: WString - локализованное название секции
 bindNames: table - таблица со списком биндов, индексация [0..],
  значения: string - идентификатор бинда

-- пример:
local section = binding.GetSection( sysSectionName )
if section then
 LogInfo( "  sysName: ", section.sysName )
 LogInfo( "  name: ", debugCommon.FromWString( section.name ) )
 for i = 0, GetTableSize( section.bindNames ) - 1 do
  local sysBindName = section.bindNames[i]
  LogInfo( "   bind ", i, ": ", sysBindName )
 end
end
