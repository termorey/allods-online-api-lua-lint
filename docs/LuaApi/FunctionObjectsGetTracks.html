objects.GetTracks()

Получить список отслеживаемых объектов Track в зоне действия аватар.

-- библиотека:
objects

-- объявление:
function GetTracks()

-- параметры:
нет

-- возвращаемое значение:
table of ObjectId - индексированный с 0 список таких объектов

-- пример:
local tracks = objects.GetTracks()
for i, trackId in pairs( tracks ) do
  LogInfo( "trackId: ", trackId )
end
