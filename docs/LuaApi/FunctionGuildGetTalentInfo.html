На главную

guild.GetTalentInfo( field, row, column )

Получить информацию о таланте гильдии из поля талантов.

-- система:
UIState/GameState

-- библиотека:
guild

-- объявление:
function GetTalentInfo( field, row, column )

-- параметры:
field: number (int) - номер поля
row: number (int) - номер строки
column: number (int) - номер столбца

-- возвращаемые значения
table / nil - таблица с информацией (nil в случае, если таланта в данной ячейке не существует)
  field: int - номер поля таланта
  row: int - номер строки таланта
  column: int - номер столбца таланта
  canUpdate: boolean - возможно ли изучить/проапдейтить данный талант
  isStart: boolean - стартовый талант
  isStored: boolean - поставлен на изучение
  isLocked: boolean - нельзя изучать
  isEmpty: boolean - является ли клетка пустой
  isLearned: boolean - выучена ли клетка таланта
  groupName: string or nil - группа талантов
  groupQuality: number (int) - качество
  abilityId: GuildAbilityId or nil - абилка

-- пример
local talentInfo = guild.GetTalentInfo( field, row, column )
