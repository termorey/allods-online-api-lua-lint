changeRoom.GetCosts()

Возвращает цены на услуги парикмахерской, выраженные в альтернативной валюте (её идентификатор ticketCurrencyId).

-- система
функция реализована в UI

-- библиотека:
changeRoom

-- объявление:
function GetCosts()

-- параметры:
нет

-- возвращаемые значения:
table or nil - если нет ошибки, то таблица с полями
  ticketCurrencyId: CurrencyId or nil - если нет ошибки, то идентификатор ресурса альтернативной валюты
  sexChangeCost: number (integer) - цена изменения пола
  petChangeCost: number (integer) - цена изменения облика пета
  petNameChangeCost: number (integer) - цена изменения имени пета
  petFacialChangeCost: number (integer) - цена изменения внешнего вида пета
  whiteList: table (VariationsTable) - таблица, описывающая цены изменения вариаций из белого списка (обычные)
  greyList: table (VariationsTable) - таблица, описывающая цены изменения вариаций из серого списка (уникальные)

VariationsTable - поля таблицы:
  skin: number (integer) - цена изменения текстуры кожи
  skinColor: number (integer) - цена изменения цвета кожи
  petColor: number (integer) - цена изменения цвета пета
  hairColor: number (integer) - цена изменения цвета волос
  hair: number (integer) - цена изменения волос
  facial: number (integer) - цена изменения дополнительных лицевых признаков
  face: number (integer) - цена изменения лица
  body: number (integer) - цена изменения строения тела
  additional: number (integer) - цена изменения дополнительных параметров


-- примеры:
local costs = changeRoom.GetCosts()
local petUnicalCost = costs.greyList.petColor
