avatar.GetCurrencyValue(currencyId)

В игре встречаются альтернативные валюты. Этот метод возвращает её количество у игрока.

-- система:
GameState

-- библиотека:
avatar

-- объявление:
function GetCurrencyValue( currencyId )

-- параметры:
currencyId: CurrencyId or nil - идентификатор ресурса альтернативной валюты.

-- возвращаемые значения:
table or nil - если нет ошибки, то таблица с полями:
  value: number( integer ) - количество такой валюты у игрока
  maxValue: number( integer ) or nil - максимально возможное количество такой валюты у игрока, если такая величина задана

-- пример:
local myrrhId = avatar.GetCurrencyId( "myrrh" )
if myrrhId then
  local info = myrrhId:GetInfo()
  if info then
    local currentValue = info.value
  end
end
