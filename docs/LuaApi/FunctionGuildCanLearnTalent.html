На главную

guild.CanLearnTalent( field, row, column, talentType )

Возвращает возможность и условия изучения таланта гильдии.

-- библиотека:
guild

-- объявление:
function CanLearnTalent( field, row, column, talentType )

-- параметры:
field: number (int) - номер поля
row: number (int) - номер строки
column: number (int) - номер столбца
talentType: number (ENUM_TalentType) or nil - тип таланата, по-умолчанию ENUM_TalentType_GUILD

-- возвращаемые значения
table or nil - таблица с полями:
  result: Boolean - true если изучение таланта доступно
  requirements: table or nil - требования для изучения таланта, таблица с полями:
    nearFieldClosed: Boolean - true если рядом нет открытых талантов
    talentsPoint: Number(int) - доступное количество очков талантов
    talentsPointRequired: Number(int) - требуемое количество очков талантов
    experience: Number(int) - доступное количество опыта
    experienceRequired: Number(int) - требуемое количество опыта
    requiredCurrency: CurrencyId or nil - требуемая альтвалюта

-- пример
local resultForGuild = guild.CanLearnTalent( 0, 4, 5 ) -- проверить условия изучения вехи гильдии
local resultForBoss = guild.CanLearnTalent( 0, 0, 1, ENUM_TalentType_BOSS ) -- проверить условия изучения второго таланта босса гильдии
