guildBossLib.GetInfo()

Возвращает сводную информацию о боссе гильдии

-- библиотека:
guildBossLib

-- объявление:
function GetInfo()

-- параметры:
нет

-- возвращаемое значение:
table or nil -- таблица с полями:
  experience: Number(int) -- накопленный опыт босса гильдии
  gearScore: Number(int) -- текущая сила босса гильдии
  name: WString or nil -- нарицательное имя босса гильдии
  description: WString -- художественное описание босса гильдии
  currentState: Number(int) or nil -- индекс текущей стадии прогресса босса гильдии
  states: table or nil -- список стадий прогресса боса, индексируется с 1, каждый элемент которого таблица с полями:
    experienceLimit: Number(int) -- предел накапливаемого опыта босса гильдии на данной стадии
    availableAbilitySlots: table of String(ENUM_GuildBossAbilitySlot) or nil -- список доступных слотов способностей босса гильдии

-- пример:
local guildBossInfo = guildBossLib.IsExist() and guildBossLib.GetInfo()
