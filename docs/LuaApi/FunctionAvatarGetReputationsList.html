На главную

avatar.GetReputationsList()

Получить у аватара список открытых репутаций.

-- библиотека:
avatar

-- объявление:
function GetReputationsList()

-- параметры:
нет

-- возвращаемые значения
table:
  key: number (int) [0..] - индекс репутации в списке
  value: table - таблица с информацией о репутации:
    factionId: FactionId - фракция
    rep: number (int) - текущее значение репутации
    level: number (enum REPUTATION_LEVEL_...) - текущий уровень репутации
    nextLevel: number (enum REPUTATION_LEVEL_...) - следующий либо максимальный уровень репутации
    repCurrLevel: number (int) / nil - значение репутации для получения текущего уровня репутации (если есть, иначе nil)
    repNextLevel: number (int) / nil - значение репутации для получения следующего уровня репутации (если есть, иначе nil)


-- пример
local reputationsList = avatar.GetReputationsList()
