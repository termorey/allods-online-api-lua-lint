function GetInfo()

нет

table - таблица с ячейками

  totalStages: Number(int) - общее количество возможных ступеней
  stageProgress: Number(float) - прогресс стадии
  sumToNextStage: Number(float) - сумма до следующей стадии
  totalProgress: Number(float) - общий прогресс
  totalSum: Number(float) - общая сумма

-- пример
local premiumInfo = premiumLib.GetInfo()
